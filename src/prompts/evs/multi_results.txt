```python
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Load the dataset
file_path = './data/evs.csv'
df = pd.read_csv(file_path)

# Clean the dataset
# Check for missing values and drop rows with missing values in relevant columns
df_cleaned = df.dropna(subset=['Electric Vehicle Type', 'Electric Range'])

# Step 1: Group data by Electric Vehicle Type and calculate summary statistics
range_summary = df_cleaned.groupby('Electric Vehicle Type')['Electric Range'].describe()
print("\nSummary Statistics of Electric Range by Vehicle Type:")
print(range_summary)

# Step 2: Visualize the distribution of Electric Range by Electric Vehicle Type
plt.figure(figsize=(12, 6))

# Boxplot to show distribution of electric range
sns.boxplot(data=df_cleaned, x='Electric Vehicle Type', y='Electric Range')
plt.xticks(rotation=45)
plt.title('Distribution of Electric Vehicle Range by Type')
plt.ylabel('Electric Range (miles)')
plt.xlabel('Electric Vehicle Type')
plt.grid(axis='y')

# Show the box plot
plt.tight_layout()
plt.show()

# Step 3: Create side-by-side histograms for each Electric Vehicle Type
# Define the list of unique Electric Vehicle Types
vehicle_types = df_cleaned['Electric Vehicle Type'].unique()

# Set up the matplotlib figure and axes for subplots
num_types = len(vehicle_types)
fig, axes = plt.subplots(nrows=1, ncols=num_types, figsize=(20, 6), sharey=True)

# Create a histogram for each Electric Vehicle Type
for ax, vehicle_type in zip(axes, vehicle_types):
    sns.histplot(df_cleaned[df_cleaned['Electric Vehicle Type'] == vehicle_type],
                 x='Electric Range', bins=30, ax=ax, color='skyblue')
    ax.set_title(vehicle_type)
    ax.set_xlabel('Electric Range (miles)')
    ax.set_ylabel('Count')
    ax.grid(axis='y')

# Add an overall title and adjust layout
plt.suptitle('Distribution of Electric Vehicle Range by Type', fontsize=16)
plt.tight_layout(rect=[0, 0, 1, 0.95])  # Adjusts the layout to make room for the title

# Show the plot
plt.show()
```
